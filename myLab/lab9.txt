/*
1. Написать функцию, выводящую на консоль имя вызвавшей её функции.

2. Написать функцию, имеющую параметры x и y и возвращающую функцию, которая всегда возвращает x+y.

3. Написать функцию makeAccumulator без параметров, которая создаёт и возвращает функцию f с одним 
   параметром. Эта функция f должна возвращать сумму всех аргументов, которые она видела за 
   всё время выполнения скрипта, нарастающим итогом. То есть первый вызов f(3) должен вернуть 3, 
   второй вызов f(2) - вернуть 5 и т.д.

*/

// task 1
function firstFunc(){
  return  firstFunc.caller.name;
}
function secondFunc(){
  return  firstFunc();
}

console.log(secondFunc())


// task 2
function func1(x, y){
  return () => x + y;
}
var func2 = func1(3, 2);
console.log(func2())
console.log(func2(2, 3))


// task 3
function makeAccumulator(){
  var sum = 0;
  return (...args) => {
    for(i in args){
      sum += args[i];
    }
    return sum;
  };
}
var f = makeAccumulator();
console.log(f(3));
console.log(f(2));
console.log(f(4, 1));
