/*
2. Написать программу, выводящую все подстроки строки
Пример : dog
Вывод : d, do, dog, o, og, g
*/

var str = 'dog';
var resultStr = '';
for(var i = 0; i < str.length ; i++){
  var lex = ''
  for(var y = i+1; y <= str.length ; y++){
    
    lex += str.substring(i, y) + ', ';
    
  }
  resultStr += lex + ', '
  
}
alert(resultStr);
