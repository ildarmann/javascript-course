function welcome(){
	arguments.join = [].join
	return 'Welcome ' + arguments.join();
}

 this.name = name;
 this.sayHello = (friends) => {
 	 friends.forEach(elem => console.log(this.name + ' say hello to ' + elem));
 }
}
